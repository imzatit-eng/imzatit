FROM node:18-alpine

WORKDIR /app

# تثبيت المتطلبات
COPY package*.json ./
RUN npm ci --only=production

# نسخ الكود
COPY . .

# إنشاء المستخدم
RUN addgroup -g 1001 -S nodejs
RUN adduser -S imzatit -u 1001
USER imzatit

# المنفذ
EXPOSE 5000

# الأمر التشغيلي
CMD ["node", "src/server.js"]
